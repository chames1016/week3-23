async function syncToSheet(payload) { if (APPS_SCRIPT_URL.includes("PASTE")) { console.warn("請設定 APPS_SCRIPT_URL"); return; } try { const response = await fetch(APPS_SCRIPT_URL, { method: "POST", headers: { "Content-Type": "text/plain" }, body: JSON.stringify(payload) }); const text = await response.text(); if (!response.ok) { console.error(`❌ 同步失敗 HTTP ${response.status}: ${text}`); showToast(`雲端同步失敗 (${response.status})`, true); return; } // 檢查後端係咪回傳咗業務錯誤 try { const data = JSON.parse(text); if (data.status === 'error') { console.error('❌ 後端回傳錯誤:', data.message); showToast('同步失敗: ' + (data.message || '未知錯誤'), true); return; } } catch (e) { // 唔係 JSON，唔理 } console.log("📤 已同步", payload); } catch (e) { console.error("同步網路錯誤", e); showToast("雲端同步失敗，已存於本地", true); } }
